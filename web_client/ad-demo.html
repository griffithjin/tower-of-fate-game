基础用法

在HTML中添加广告容器，然后通过JavaScript触发展示

// 1. 展示广告 window.adSystem.showAd('victory-screen'); // 2. 隐藏广告 window.adSystem.hideAd('victory-screen'); // 3. 获取统计数据 const stats = window.adSystem.getStats(); console.log(stats.revenue); // 今日收入 // 4. 创建广告活动 window.adSystem.addCampaign({ name: '新手推广', type: 'banner', placements: ['main-menu-banner'], content: { image: '/ads/banner.jpg', link: '/shop' } });
